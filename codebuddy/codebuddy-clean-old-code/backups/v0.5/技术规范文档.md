#### 4.2.1 输入验证
```python
def validate_stock_code(code):
    """
    验证股票代码格式
    
    Args:
        code: 股票代码
        
    Returns:
        bool: 是否有效
    """
    if not code or not isinstance(code, str):
        return False
    
    # 验证格式：6位数字
    if len(code) != 6 or not code.isdigit():
        return False
    
    # 验证市场代码
    market = code[:2]
    if market not in ['60', '00', '30']:
        return False
    
    return True
```

#### 4.2.2 数据清洗
```python
def clean_stock_data(data):
    """
    清洗股票数据
    
    Args:
        data: 原始数据
        
    Returns:
        dict: 清洗后的数据
    """
    cleaned = {}
    
    # 移除空值
    for key, value in data.items():
        if value is not None and value != '':
            cleaned[key] = value
    
    # 验证必填字段
    required_fields = ['code', 'name', 'market']
    for field in required_fields:
        if field not in cleaned:
            raise ValueError(f"缺少必填字段: {field}")
    
    return cleaned
```

import sqlite3
from contextlib import contextmanager

@contextmanager
def get_db_connection():
    conn = sqlite3.connect('data/stock.db')
    conn.row_factory = sqlite3.Row
    try:
        yield conn
    finally:
        conn.close()

# 批量操作
def batch_insert_stocks(stocks):
    with get_db_connection() as conn:
        cursor = conn.cursor()
        cursor.executemany(
            "INSERT INTO stocks (code, name, market) VALUES (?, ?, ?)",
            [(s['code'], s['name'], s['market']) for s in stocks]
        )
        conn.commit()

# 简单的内存缓存
import time

class SimpleCache:
    def __init__(self, timeout=300):
        self.cache = {}
        self.timeout = timeout
    
    def get(self, key):
        if key in self.cache:
            value, timestamp = self.cache[key]
            if time.time() - timestamp < self.timeout:
                return value
            else:
                del self.cache[key]
        return None
    
    def set(self, key, value):
        self.cache[key] = (value, time.time())

import shutil
import datetime

def backup_database():
    """备份数据库"""
    source = 'data/stock.db'
    timestamp = datetime.datetime.now().strftime('%Y%m%d_%H%M%S')
    backup_dir = 'backups'
    backup_file = f'{backup_dir}/stock_db_{timestamp}.db'
    
    # 创建备份目录
    os.makedirs(backup_dir, exist_ok=True)
    
    # 备份文件
    shutil.copy2(source, backup_file)
    
    # 清理旧备份（保留最近7天）
    cleanup_old_backups(backup_dir, days=7)
    
    return backup_file

def cleanup_old_backups(backup_dir, days=7):
    """清理旧备份文件"""
    cutoff_time = datetime.datetime.now() - datetime.timedelta(days=days)
    
    for filename in os.listdir(backup_dir):
        if filename.startswith('stock_db_') and filename.endswith('.db'):
            file_path = os.path.join(backup_dir, filename)
            file_time = datetime.datetime.fromtimestamp(os.path.getmtime(file_path))
            
            if file_time < cutoff_time:
                os.remove(file_path)
                print(f'删除旧备份: {filename}')
```

def calculate_profit_loss(current_price, cost_price, quantity):
    """
    计算股票盈亏
    
    Args:
        current_price (float): 当前价格
        cost_price (float): 成本价格
        quantity (int): 持有数量
        
    Returns:
        dict: 包含盈亏信息的字典
            - profit_loss (float): 盈亏金额
            - profit_loss_rate (float): 盈亏比例
            - is_profit (bool): 是否盈利
    
    Example:
        >>> calculate_profit_loss(10.5, 10.0, 100)
        {'profit_loss': 50.0, 'profit_loss_rate': 0.05, 'is_profit': True}
    """
    profit_loss = (current_price - cost_price) * quantity
    profit_loss_rate = (current_price - cost_price) / cost_price
    is_profit = profit_loss > 0
    
    return {
        'profit_loss': profit_loss,
        'profit_loss_rate': profit_loss_rate,
        'is_profit': is_profit
    }
```
